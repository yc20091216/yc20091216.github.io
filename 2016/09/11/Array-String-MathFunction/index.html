数组方法里push、pop、shift、unshift、join、split分别是什么作用?

  • push()方法: 在数组的最后添加一个或者多个元素,并且返回当前的length。
    var a = [1,2,3,4];
a.push(5);
console.log(a);      // a = [1,2,3,4,5]
console.log(a.length);  // 5
a.push(6,7,8);
console.log(a);    // a = [1,2,3,4,5,6,7,8]
console.log(a.length);  // 8
  • pop()方法:用于删除并返回数组的最后一个元素。
    var a = [1,2,3,4];
a.pop();            //返回删除的值 4；
console.log(a);     //  a = [1,2,3]
console.log(a.length);  //  3
  • shift()方法：用于删除并返回数组的最前面一个元素。
    var a = [1,2,3,4];
a.shift();            //返回删除的值 1；
console.log(a);     //  a = [2,3,4]
console.log(a.length);  //  3
  • unshift()方法：在数组的最前面添加一个或者多个元素，并返回当前的length。
    var a = [1,2,3,4];
a.unshift(0);
console.log(a);      // a = [0,1,2,3,4]
console.log(a.length);  // 5
a.unshift(-3,-2,-1);
console.log(a);    // a = [-3,-2,-1,0,1,2,3,4]
console.log(a.length);  // 8
  • jion()方法：把数组元素（对象调用其toString()方法）使用参数作为连接符连接成一字符串，不会修改原数组内容。
    var a = [1,2,3,4];
console.log(a.join());//不填写参数时默认逗号作为连接符 "1,2,3,4"
console.log(a.join("")); //参数为空字符串时输出  "1234"
console.log(a.join("-"));  // "1-2-3-4"
console.log(typeof a);  //  object
console.log(a);       // [1,2,3,4]

  • split()方法：把一个字符串分割成字符串数组。默认不输入参数时字符串整体作为一项数组。有两个参数，第一个参数作为分隔符来分隔元素，第二个是分割出来的数组项个数。

    var a = "1a2b3c4d";
console.log(a.split());        // ["1a2b3c4d"]
console.log(a.split("b"));     //["1a2", "3c4d5e"]
console.log(a.split(""));      //["1", "a", "2", "b", "3", "c", "4", "d", "5", "e"]
console.log(a.split("",3));    //["1","a","2"]   
console.log(typeof a);        //"string"

  • reverse()方法：颠倒数组中元素的顺序，修改并返回改变后的数组。

    var a = [1,2,3,4,5];
a.reverse();
console.log(a);   //  [5,4,3,2,1];

  • slice()方法：提取原数组的一部分，返回一个新数组，原数组不变。

    var a = [1,2,3,4,5];
console.log(a.slice());  // [1,2,3,4,5]
console.log(a.slice(3));  //[4,5]
console.log(a.slice(0,3)); //[1,2,3]
console.log(a.slice(-3,-1));  //[3,4] 参数为负数表示倒数位置。
console.log(a);   //[1,2,3,4,5]  原数组不变。

  • map()方法：对数组的所有成员依次调用一个函数，根据函数结果返回一个新数组,原数组不变。

    var a = [1,2,3,4,5];
console.log(a.map(function ( n ){
               return n + 1;
}));    //[2,3,4,5,6]
 console.log(a)；  // [1,2,3,4,5]

5. 对象obj有个className属性，里面的值为的是空格分割的字符串(和html元素的class特性类似)，写addClass、removeClass函数，有如下功能：
var obj = {
    className: 'open menu'
}

function addClass(obj, value) {
    var tempArray = obj.className.split(" ");
    for (var i = 0; i < tempArray.length; i++) {
        if (tempArray[i] === value) {
            return obj.className
        }
        tempArray.push(value);
        return obj.className = tempArray.join(" ");
    }
}


function removeClass(obj, value) {
    var tempArray = obj.className.split(" ");
    for (var i = 0; i < tempArray.length; i++) {
        if (tempArray[i] === value) {
            tempArray.splice(i, 1);
        }

        return obj.className = tempArray.join(" ");
    }
}
addClass(obj, 'new') // obj.className='open menu new'
addClass(obj, 'open')  // 因为open已经存在，所以不会再次添加open
addClass(obj, 'me') // me不存在，所以 obj.className变为'open menu new me'
console.log(obj.className)  // "open menu new me"

removeClass(obj, 'open') // 去掉obj.className里面的 open，变成'menu new me'
removeClass(obj, 'blabla')  // 因为blabla不存在，所以此操作无任何影响
6. 写一个camelize函数，把my-short-string形式的字符串转化成myShortString形式的字符串，如:
camelize("background-color") == 'backgroundColor'
camelize("list-style-image") == 'listStyleImage'
function camelize(str) {
    var tempArray = str.split("");
    for (var i = 0; i < tempArray.length; i++) {
        if (tempArray[i] === "-") {
            tempArray.splice(i, 1);
        }
    }
    return tempArray.join("");
}

var john = { name: "John Smith", age: 23 }
var mary = { name: "Mary Key", age: 18 }
var bob = { name: "Bob-small", age: 6 }
var people = [ john, mary, bob ]

function ageSort(people) {
    people.sort(function(a, b) {
        if (a.age < b.age) {         //    return    a.age - b.age
            return -1;
        }
        if (a.age > b.age) {
            return 1;
        }
    })
}

ageSort(people) // [ bob, mary, john ]

arr = ["a",3,4,true, -1, 2, "b"]
function isNumeric(el) {
    return typeof el === 'number';
}

function filter(arr,fn){
      for(var i = arr.length-1; i >= 0 ; i--){
        if(!fn(arr[i])){
          arr.splice(i,1);
        }
      }return arr
}        
arr = filter(arr, isNumeric) ; // arr = [3,4,-1, 2],  过滤出数字
arr = filter(arr, function(val) { return  typeof val === "number" && val > 0 });  // arr = [3,4,2] 过滤出大于0的整数

function getRandStr(len) {
    var scope = "ABCDEFGHIZKLMNOPQRSTUVWXYZ";
    scope += "abcdefghizklmnopqrstuvwxyz";
