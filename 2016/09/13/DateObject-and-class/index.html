基础类型有哪些？复杂类型有哪些？有什么特征？

  • 基本数据类型：Undefined、Null、Boolean、Number、String(简单的数据段)。
    • 基本数据类型是按值访问的，因为可以操作保存在变量中的实际的值。
  • 复杂类型(引用类型)：Object(可能由多个值构成的对象)。
    • 复杂类型的值是保存在内存中的对象。

  • 特征：

    • 只能给引用类型值动态的添加属性，以便将来使用。
      var student = {};
      student.name = “Jhon”;
      console.log(student.name); // Jhon
      var name = “Jhon”
      name.age = 18;
      console.log(name.age) // Undefined;

    • 复制变量值：

      • 如果从一个变量向另一个变量复制基本类型的值，会在变量对象上创建一个新值，然后把该值复制到新变量分配的位置上，这两个变量可以参与任何操作而不会互相影响。
        var count1 = 18;
        var count2 = count1;
        count1 = 20;
        console.log(count1); // 20
        console.log(count2); // 18
      • 当从一个变量向另一个变量复制引用类型的值时，这个值得副本实际上是一个指针，指向同一个对象，因此改变一个变量就会影响到另一个变量。
        var count1 = {first:18};
        var count2 = count1;
        count1.first = 30;
        console.log(count1.first); // 30
        console.log(count2.first); // 30

如下代码的输出? 为什么?

var obj1 = {a:1, b:2};
var obj2 = {a:1, b:2};
console.log(obj1 == obj2); //false,因为是两个不同的对象，跟它们的值相同没什么关系。
console.log(obj1 = obj2); //将obj2赋值给obj1。这时obj1指向obj2内存中对象的值。
console.log(obj1 == obj2); // 指向同一对象，所以当然是true。

写一个函数getIntv，获取从当前时间到指定日期的间隔时间？
var str = getIntv("2016-01-08");
console.log(str);  // 距除夕还有 20 天 15 小时 20 分 10 秒

function getIntv(time) {
    var now = new Date(),
        appoint = new Date(time),
        result = appoint - now;
    var days = Math.floor(result / (24 * 60 * 60 * 1000)),
        hours = Math.floor((result - days * 24 * 60 * 60 * 1000) / (60 * 60 * 1000)),
        minutes = Math.floor((result - hours * 60 * 60 * 1000) / (60 * 1000)),
        seconds = Math.floor((result - minutes * 60 * 1000) / 1000);
    var str = "距离指定时间还有" + days + "天" + hours + "小时" + minutes + "分钟" + seconds + "秒";
    return str;
}

function getChsDate(val) {
    var valArr = val.split("-"),
        tempArr = ["零", "一", "二", "三", "四", "五", "六", "七", "八", "九"],
        year = [];
    for (var i = 0; i < valArr[0].length; i++) {
        year = year + tempArr[valArr[0][i]];
    }

    function getDate(date) {
        switch (true) {
            case date < 10:
                return tempArr[date[1]];

            case date === 10:
                return "十";

            case date > 10 && date < 20:
                return "十" + tempArr[date[1]];

            case date > 20 && date < 30:
                return "二十" + tempArr[date[1]];

            default:
                return;
        }
    }
    return year + "年" + getDate(valArr[1]) + "月" + getDate(valArr[2]) + "日";
}

function getLastNDays(n){
   var now = new Date(),
       days = n * 24 * 60 * 60 * 1000,
       date =  new Date(now - days);
   var temp = [];
       temp[0] = date.getUTCFullYear();
       temp[1] = date.getUTCMonth();
       temp[2] = date.getUTCDate();
  return temp.join("-")
}

function fn(mun) {
switch (true) {
case mun === 1:
    return 1;
case mun === 2:
    return 2;
default:
    return fn(mun - 1) + fn(mun - 2);
}

写一个json对象深拷贝的方法，json对象可以多层嵌套，值可以是字符串、数字、布尔、json对象中的任意项
function jsonCopy(json) {
    var newJson = {};
    for (var key in json) {
        if (json[key] === "object") {
            newJson[key] = jsonCopy(json[key]);
        } else {
            newJson[key] = json[key];
        }
    }
    return newJson;
}

var json = {
    "name": "json",
    "age": 18,
    "address": {
        "area": "China",
        "city": "FJ",
    }
};
console.log(jsonCopy(json));
